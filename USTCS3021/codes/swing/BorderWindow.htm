import java.awt.*;
import java.awt.event.*;

import javax.swing.*;

public class BorderWindow extends JFrame {

public static void main(String args[]) {
        BorderWindow frame = new BorderWindow();
        frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        frame.setVisible(true);
    }
public BorderWindow() {
    setSize(400,200);
    setTitle("BorderLayout");

         add(new JButton("Button 1 (NORTH)"),
                        BorderLayout.NORTH);
         add(new JButton("2 (CENTER)"),
                        BorderLayout.CENTER);
         add(new JButton("Button 3 (WEST)"),
                        BorderLayout.WEST);
         add(new JButton("Long-Named Button 4 (SOUTH)"),
                        BorderLayout.SOUTH);
         add(new JButton("Button 5 (EAST)"),
                        BorderLayout.EAST);


    }


}
