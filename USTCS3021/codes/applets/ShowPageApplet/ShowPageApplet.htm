import java.awt.*;
import java.awt.event.*;
import javax.swing.*;
import javax.swing.border.*;
import java.net.*;

public class ShowPageApplet extends JApplet
   implements ActionListener
{  public void init()
   {
       // UI components
      Container contentPane = getContentPane();
  JLabel lab = new JLabel("Password required", SwingConstants.CENTER);
      contentPane.add("North", lab);

      JPanel p1 = new JPanel();
      p1.setLayout(new GridLayout(2, 2));
      p1.add(new JLabel("User name:"));
      p1.add(username = new JTextField("your account name"));
      p1.add(new JLabel("Password:"));
      p1.add(password = new JPasswordField("***"));
      contentPane.add("Center", p1);


      Border b1 = BorderFactory.createEmptyBorder(10, 10, 10, 10);
      p1.setBorder( b1 );

      Panel p3 = new Panel();
      okButton = addButton(p3, "Ok");
      clearButton = addButton(p3, "Clear");
      gobackButton = addButton(p3, "Go Back");
      contentPane.add("South", p3);

   }

    // helper method
   JButton addButton(Container c, String name)
   {  JButton button = new JButton(name);
      button.addActionListener(this);
      c.add(button);
      return button;
   }

    // handling event
   public void actionPerformed(ActionEvent evt)
   {  Object source = evt.getSource();
      if(source == okButton)
      {
         char[] psswd = password.getPassword();
         if ( psswd.length ==3 &&
           psswd[0] == '2' && psswd[1] == '0' && psswd[2]=='1' )
         {
            try {
        // *** ask browser to show a page *******
        String s="http://www.yahoo.com/";
        URL u = new URL(s);
        getAppletContext().showDocument(u);
            }
            catch( MalformedURLException e) {}
         }
   else {
       password.setText("");
       username.setText("");
         }
      }
      else if (source == clearButton) {
       password.setText("");
       username.setText("");
      }
      else if (source == gobackButton) {
            try {
        // *** ask browser to show a page *******
        String s="http://www.cs.ust.hk/course/comp201";
        URL u = new URL(s);
        getAppletContext().showDocument(u);
            }
            catch( MalformedURLException e) {}
      }

   }

   private JMenuItem connectItem;
   private JMenuItem exitItem;
   private JButton okButton;
   private JButton clearButton;
   private JButton gobackButton;
   private JPasswordField password;
   private JTextField username;
}
