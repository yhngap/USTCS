import java.awt.*;
import java.awt.event.*;
import javax.swing.*;
import java.net.*;



public class ImageApplet extends JApplet
{  public void init()
   {  JFrame frame = new RadioButtonFrame();      
           frame.pack();  
      frame.show();
   }
}

class RadioButtonDemo extends JPanel {
    static JFrame frame;

    static String birdString = "Bird";
    static String catString = "Cat";
    static String dogString = "Dog";
    static String rabbitString = "Rabbit";
    static String pigString = "Pig";

    Image bird, cat, dog, rabbit, pig;
    JLabel picture;

    public RadioButtonDemo() {
  // load images
  bird = loadImage( birdString );
  cat = loadImage( catString );
  dog = loadImage( dogString );
  rabbit = loadImage( rabbitString );
  pig = loadImage( pigString );

        // Create the radio buttons.
        //XXX: In 1.0.2 only, must use FixedJRadioButton instead of
        //XXX: JRadioButton if you want mnemonics to work.
        //XXX: See ui/swing/workaround.html for details.
        JRadioButton birdButton = new JRadioButton(birdString);
        birdButton.setMnemonic('b');
        birdButton.setActionCommand(birdString);
        birdButton.setSelected(true);

        JRadioButton catButton = new JRadioButton(catString);
        catButton.setMnemonic('c');
        catButton.setActionCommand(catString);

        JRadioButton dogButton = new JRadioButton(dogString);
        dogButton.setMnemonic('d');
        dogButton.setActionCommand(dogString);

        JRadioButton rabbitButton = new JRadioButton(rabbitString);
        rabbitButton.setMnemonic('r');
        rabbitButton.setActionCommand(rabbitString);

        JRadioButton pigButton = new JRadioButton(pigString);
        pigButton.setMnemonic('p');
        pigButton.setActionCommand(pigString);

        // Group the radio buttons.
        ButtonGroup group = new ButtonGroup();
        group.add(birdButton);
        group.add(catButton);
        group.add(dogButton);
        group.add(rabbitButton);
        group.add(pigButton);

        // Register a listener for the radio buttons.
        RadioListener myListener = new RadioListener();
        birdButton.addActionListener(myListener);
        catButton.addActionListener(myListener);
        dogButton.addActionListener(myListener);
        rabbitButton.addActionListener(myListener);
        pigButton.addActionListener(myListener);

        // Set up the picture label
        picture = new JLabel(new ImageIcon(bird) );

        // The preferred size is hard-coded to be the width of the
        // widest image and the height of the tallest image.
        // A real program would compute this.
        picture.setPreferredSize(new Dimension(177, 122));


        // Put the radio buttons in a column in a panel
        JPanel radioPanel = new JPanel();
        radioPanel.setLayout(new GridLayout(0, 1));
        radioPanel.add(birdButton);
        radioPanel.add(catButton);
        radioPanel.add(dogButton);
        radioPanel.add(rabbitButton);
        radioPanel.add(pigButton);

        setLayout(new BorderLayout());
        add(radioPanel, BorderLayout.WEST);
        add(picture, BorderLayout.CENTER);
        setBorder(BorderFactory.createEmptyBorder(20,20,20,20));
    }


    /** Listens to the radio buttons. */
    class RadioListener implements ActionListener {
        public void actionPerformed(ActionEvent e) {
      String s = e.getActionCommand();

            if (s.equals( birdString ) )
              picture.setIcon(new ImageIcon(bird) );
            else if (s.equals( catString ) )
              picture.setIcon(new ImageIcon(cat) );
            else if (s.equals( dogString ) )
              picture.setIcon(new ImageIcon(dog) );
            else if (s.equals( rabbitString ) )
              picture.setIcon(new ImageIcon(rabbit) );
            else if (s.equals( pigString ) )
              picture.setIcon(new ImageIcon(pig) );
        }
    }

   private Image loadImage (String s)
   {

      // get address of image
      URL url =  RadioButtonDemo.class.getResource("images/"
                                          +  s
                                          + ".gif");

      // load image
      Image image = Toolkit.getDefaultToolkit().getImage( url );

      return image;
   }

}

class RadioButtonFrame extends JFrame
{   RadioButtonFrame()
    { setTitle("RadioButtonDemo");

      // no Window Listener

      getContentPane().add(new RadioButtonDemo(), BorderLayout.CENTER);
     }
}
