public class Swapping
{  public static void main(String argv[])
   {  int[] a = {1,2,3};
        for (int element: a)
          System.out.print(element+" ");

      System.out.print("\n");
        swap(a,0,2);
      for (int element: a)
          System.out.print(element+" ");
    System.out.print("\n");
   }

   public static void swap(int[] a, int i, int j)
   { int temp = a[j];
     a[j]=a[i];
     a[i]=temp;
   }
}
